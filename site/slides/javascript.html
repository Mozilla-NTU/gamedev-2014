function: 
var fun = function(foo){
  console.log("hello", foo);
}
fun("NTU");  // logs "hello NTU"

function fun2(foo){
  console.log("hello", foo);
}
fun2("NTU"); // logs "hello NTU"
Arrays: 

var arr = [1,"a", 3, null];
arr[0];     => 1
arr[3];     => null

arr[4];     => undefined
arr[-1];    => undefined
arr.length;  => 4

arr.push("hello");
arr.length;  => 5 
console.log(arr); // prints [1, "a", 3, null, "hello"]
